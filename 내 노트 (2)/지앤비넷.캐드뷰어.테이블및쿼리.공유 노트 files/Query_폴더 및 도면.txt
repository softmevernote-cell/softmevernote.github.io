
SELECT A.FOLID,A.FOLNM,A.FOLPT,A.FOLLV,A.SEQ, B.DOCNO, B.DOCVR, B.DOCNM, B.DOCNUMBER, B.PLANTCODE, B.PIPEDOC_YN, B.PIPE_YN FROM IDS_FOLDER A, IDS_DOC B 
WHERE A.FOLID = B.FOLID(+)
AND A.FOLID IN (SELECT FOLID FROM IDS_ACL_FOLDER WHERE ACLTP = '001' GROUP BY FOLID)
ORDER BY A.FOLLV, A.PLANTCODE, A.FOLPT, A.SEQ
